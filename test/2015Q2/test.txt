	function processShortcut(event) {
			if (isIE)
		{
			event = window.event;
			event.target = event.srcElement;
		}

		if (event.altKey || event.ctrlKey || event.metaKey)
			return;

			case "INPUT":
			case "SELECT":
			case "TEXTAREA":
				return;
		}
		switch (event.keyCode) {
			case 81: //Q
